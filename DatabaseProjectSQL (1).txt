CREATE TABLE Members
( std_id      char(9)        NOT NULL,
  std_name    varchar(15)    NOT NULL,
  std_surname varchar(15)    NOT NULL,
  std_faculty varchar(35)    NOT NULL,
  std_major   varchar(15)    NOT NULL,
  std_email   varchar(25)    NOT NULL,
  PRIMARY KEY (std_id));

CREATE TABLE Certificate 
(
  certificate_id   char(6)  NOT NULL,
  certificate_res  char(6)  NOT NULL,
  std_id           char(9)  NOT NULL,
  signed_by     varchar(15) NOT NULL,
  PRIMARY key (certificate_id),
  FOREIGN KEY (std_id) references Members (std_id));

CREATE TABLE MemebersPhone_nr
(
  Phone_nr    Char(15)     NOT NULL,
  std_id      char(9)      NOT NULL,
  PRIMARY key (Phone_nr),
  FOREIGN KEY (std_id) references Members (std_id));

CREATE table Clubs
(
  Club_id        char(4)     NOT NULL,
  Club_name      varchar(30) NOT NULL,
  club_capacity  int         NOT NULL,
  Created_date   Date        NOT NULL,
  PRIMARY KEY (Club_id)
);
 

CREATE table PartOf
(
  Club_id char(4) NOT NULL,
  std_id char(9) NOT NULL,
  FOREIGN KEY (std_id)REFERENCES Members (std_id),
  FOREIGN KEY (Club_id) REFERENCES Clubs (Club_id));

CREATE table StaffRole
(
  Role_id   char(8)     NOT NULL,
  role_name varchar(15) NOT NULL,
  PRIMARY KEY (Role_id)
);


CREATE table ClubStaff
(
  clubstaff_id char(9)     NOT  NULL,
  Pers_name     varchar(10) NOT NULL,
  Pers_surname  varchar(10) NOT NULL,
  Club_id      char(4)     NOT NULL,
  Role_id      char(8)     NOT NULL,
  PRIMARY KEY ( clubstaff_id),
  FOREIGN KEY (Club_id) REFERENCES Clubs (Club_id),
  FOREIGN KEY (Role_id) REFERENCES StaffRole (Role_id));


CREATE TABLE Activities
(
  Activity_name varchar(50) NOT NULL,
  date          Date        NOT NULL,
  hour          time    Not NULL,
  capacity      int         not NULL,
  place         varchar(15) NOT NULL,
  guest_name    varchar(15),
  Club_id       char(4)     NOT NULL,
  PRIMARY KEY   (Activity_name),
  FOREIGN KEY  (Club_id) REFERENCES Clubs(Club_id)
  );

CREATE  TABLE  part_of
(
  std_id         char(9)     NOT NULL,
  activity_name  varchar(50) NOT NULL,
  FOREIGN KEY (std_id) REFERENCES Members (std_id),
  FOREIGN KEY  (activity_name) REFERENCES Activities(activity_name)
  );
